void loop() {
  // 1. Read HTU21D (Temp & Hum)
  SensorData data = readHTUSensor();
  
  // 2. Read MAX30102 (HR & SpO2)
  // Note: This function blocks slightly to read samples
  readMAX30102Sensor(data);
  
  // 3. Print to Serial Monitor
  printSensorData(data);
  
  // 4. Send to Firebase
  if (millis() - lastFirebaseUpdate >= FIREBASE_UPDATE_INTERVAL) {
    sendToFirebase(data);
    lastFirebaseUpdate = millis();
  }
  delay(5000); // Wait 5 seconds between reads
}

void setupSensors() {

SensorData readHTUSensor() {
  SensorData data;
  data.timestamp = millis();
  
  // Read from HTU21D
  data.temperature = htu.readTemperature();
  data.humidity = htu.readHumidity();
  
  return data;
}

void readMAX30102Sensor(SensorData &data) {
  // This logic is preserved from your friend's code
  long irValue = particleSensor.getIR();

  // Heart Beat Detection
  if (checkForBeat(irValue) == true) {
    long delta = millis() - lastBeat;
    lastBeat = millis();
    beatsPerMinute = 60 / (delta / 1000.0);

    if (beatsPerMinute < 255 && beatsPerMinute > 20) {
      rates[rateSpot++] = (byte)beatsPerMinute;
      rateSpot %= RATE_SIZE;
      
      beatAvg = 0;
      for (byte x = 0 ; x < RATE_SIZE ; x++)
        beatAvg += rates[x];
      beatAvg /= RATE_SIZE;
      
      data.heartRate = beatAvg;
      validHeartRate = 1;
    }
  }

  // Simplified SpO2 Calculation
  uint32_t ir = particleSensor.getIR();
  uint32_t red = particleSensor.getRed();

  avered = avered * frate + (double)red * (1.0 - frate);
  aveir = aveir * frate + (double)ir * (1.0 - frate);
  sumredrms += (double)red * red;
  sumirrms += (double)ir * ir;

  if ((i % Num) == 0) {
    double R = (sqrt(sumredrms) / avered) / (sqrt(sumirrms) / aveir);
    data.spo2 = -19.0 * R + 99.0;
    
    if (data.spo2 >= 90 && data.spo2 <= 100) validSpO2 = 1;
    
    sumredrms = 0.0; sumirrms = 0.0; i = 0;
  }
  i++;

  // Default to 0 if invalid
  if (!validHeartRate) data.heartRate = 0;
  if (!validSpO2) data.spo2 = 0;
}

void printSensorData(const SensorData &data) {
  Serial.print("Temp: "); Serial.print(data.temperature);
  Serial.print("C | Hum: "); Serial.print(data.humidity);
  Serial.print("% | HR: "); Serial.print(data.heartRate);
  Serial.print(" | SpO2: "); Serial.println(data.spo2);
}

void sendToFirebase(const SensorData &data) {
  if (Firebase.ready()) {
    
    // Create the JSON object
    FirebaseJson json;
    json.set("temperature", data.temperature);
    json.set("humidity", data.humidity);
    json.set("heartRate", data.heartRate);
    json.set("spo2", data.spo2);
    // Add a readable timestamp
    json.set("timestamp_ms", data.timestamp);

    Serial.println("Pushing to Firebase...");

    // --- FIXED: pushJSON creates a unique ID for history ---
    // Path: /sensorReadings/-Nj8s... (unique ID)
    if (Firebase.RTDB.pushJSON(&fbdo, "/sensorReadings", &json)) {
      Serial.println("✓ Success");
    } else {
      Serial.println("✗ Error: " + fbdo.errorReason());
    }
  }
}
